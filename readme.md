# 克隆项目
git clone https://github.com/biubiubiu01/vue-antd-server.git

# 本地已经安装mysql且正常启动

# 最好是安装了navicat，打开navicat，新建一个数据库 然后右键运行sql 选择 server文件夹下面的 web_antd.sql 等待一会后即可获取该项目所有表

# 进入项目目录
cd vue-antd-server

# 安装依赖
npm install

# 运行系统
npm run serve

# 进入node服务器文件夹
cd vue-antd-server/server

# 安装依赖
npm install

# 运行node服务器
npm start
